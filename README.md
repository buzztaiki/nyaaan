# Nyaaan

## Usage
```console
% nyaaan src/main.rs
nya nyan::nyaaan;
nya nya::nya;

nya nya;
nya nyaaan;
nya nyaaan;
nya nyaaaaaan;

ny nyan() -> nyaaan::nyaaan<()> {
    nya::nyaan().nya()
}
```

```console
% nyaaan --nya "moo" --n "" < src/main.rs
moo mooo::mooooo;
moo moo::moo;

moo moo;
moo mooooo;
moo mooooo;
moo moooooooo;

mo mooo() -> mooooo::mooooo<()> {
    moo::moooo().moo()
}
```
